    Floor floor[134];                // ç»è®¡ç®—å¯å¾—ç¼ºæŸæ ‘æœ€å¤š133å±‚ï¼Œç¬¬0å±‚ä¸º1
    floor[0].start = floor[0].end = 1;

    scanf("%d%d", &N, &M);
    for (int i = 0; i < N; i++)
    {
        scanf("%lld", &defect[i]);
    }
    for (int i = 0; i < M; i++)
    {
        scanf("%lld", &target[i].data);
    }

    if (defect[0] == 1) // å¦‚æœæ ¹èŠ‚ç‚¹1ç¼ºæŸï¼Œé‚£ä¹ˆåç»­å…¨éƒ½ä¸å­˜åœ¨äº†
    {
        for (int i = 0; i < M; i++)
        {
            if (target[i].data == 1)
                printf("%d\n", 1);
            else
                printf("%d\n", 0);
        }
        return 0;
    }

    // å¼€å§‹æ„é€ ç¼ºæŸæ ‘
    int cur_fl = 1; // å½“å‰å±‚æ•°
    int df_num = 0; // å½“å‰å±‚çš„æ‰€æœ‰ç¼ºæŸç‚¹æ•°
    int flag_d = 0; // å½“å‰å°†è¦é‡åˆ°å“ªä¸ªç¼ºæŸç‚¹defect[flag_d]
    int flag_t = 0; // å½“å‰å°†è¦é‡åˆ°å“ªä¸ªç›®æ ‡ç‚¹target[flag_t].data
    long long last_nd;

    if (target[0].data == 1)
    {
        target[0].floor = 0;
        flag_t++;
    }

    for (cur_fl = 1; cur_fl < 134; cur_fl++)
    {
        floor[cur_fl].start = floor[cur_fl - 1].end + 1;                                                              // ç¡®å®šèµ·ç‚¹
        floor[cur_fl].end = floor[cur_fl].start + (floor[cur_fl - 1].end - floor[cur_fl - 1].start - df_num) * 2 + 1; // ç¡®å®šç»ˆç‚¹
        df_num = 0;                                                                                                   // é‡ç½®è¿™ä¸€è¡Œçš„ç¼ºæŸç‚¹æ•°

        int i;
        for (i = flag_d; i < N; i++)
        {
            // åˆ¤æ–­è¿™ä¸€è¡Œæœ‰å‡ ä¸ªç¼ºæŸç‚¹
            if (floor[cur_fl].start <= defect[i] && defect[i] <= floor[cur_fl].end)
            {
                df_fl[cur_fl][i - flag_d + 1] = defect[i];
                df_num++;
            }
            else
                break;
        }
        // æ­¤æ—¶ï¼Œè¿™ä¸€è¡Œçš„ç¼ºæŸç‚¹æ•°ç›®ä¸ºdf_num
        df_fl[cur_fl][0] = df_num; // è¿™ä¸€è¡Œçš„ç¼ºæŸç‚¹ä¸ªæ•°å­˜æ”¾åœ¨æ•°ç»„ç¬¬ä¸€ä½
        flag_d = i;

        int j;
        for (j = flag_t; j < M; j++)
        {
            // è®°å½•è¯¥å±‚æ‰€æœ‰ç›®æ ‡ç‚¹çš„å±‚æ•°
            if (floor[cur_fl].start <= target[j].data && target[j].data <= floor[cur_fl].end)
            {
                target[j].floor = cur_fl;
            }
            else
                break;
        }
        flag_t = j;

        if (floor[cur_fl].end >= target[M - 1].data || floor[cur_fl].end - floor[cur_fl].start + 1 == df_num)
        {
            // å¦‚æœå·²ç»æ„é€ åˆ°æœ€åä¸€ä¸ªç›®æ ‡ç‚¹ï¼Œæˆ–è€…è¿™ä¸€è¡Œå…¨æ˜¯ç¼ºæŸç‚¹ï¼Œåœæ­¢æ„é€ 
            last_nd = floor[cur_fl].end; // è®°å½•æ„é€ åˆ°çš„æœ€åä¸€ä¸ªèŠ‚ç‚¹
            break;
        }
    }

    // å¼€å§‹æœç´¢å›åˆ°æ ¹ç»“ç‚¹çš„è·¯å¾„
    for (int i = 0; i < M; i++)
    {
        if (target[i].data > last_nd)
        {
            printf("%d\n", 0);
            continue;
        } // è·¯å¾„ä¸å­˜åœ¨çš„æƒ…å†µï¼Œç›´æ¥è¾“å‡º0

        long long temp[134] = {1};
        // è¯¥ç›®æ ‡ç‚¹å±‚æ•°ä¸ºtarget[i].floor
        // åœ¨df_fl[floor]ä¸­è¿˜è¦åˆ¤æ–­å¯èƒ½ç¢°åˆ°çš„ç¼ºæŸç‚¹
        int cur_fl = target[i].floor;      // å½“å‰å±‚æ•°
        long long new_ta = target[i].data; // å½“å‰èŠ‚ç‚¹
        while (cur_fl > 0)
        {
            temp[cur_fl] = new_ta;
            long long cnt = (new_ta - floor[cur_fl].start) / 2;
            int flag = 0;
            for (int j = 1; j <= df_fl[cur_fl - 1][0]; j++)
            {
                if (cnt + j <= df_fl[cur_fl - 1][j] - floor[cur_fl - 1].start)
                {
                    break;
                }
                else
                {
                    flag = j;
                }
            }
            new_ta = floor[cur_fl - 1].start + cnt + flag;
            cur_fl--;
        }
        printf("%d", 1); // æœ€å…ˆè¾“å‡ºæ ¹èŠ‚ç‚¹
        for (int j = 1; j <= target[i].floor; j++)
        {
            printf(" %lld", temp[j]);
        }
        printf("\n");
    }

    return 0;
